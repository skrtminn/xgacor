            function updateVariant(size, color) {
                const key = size + '-' + color;
                const variant = variantData[key];
                if (!variant) return;

                // Update harga
                document.getElementById('productPrice').innerHTML = `Rp${variant.price.toLocaleString('id-ID')} <small>Rp1.000.000</small>`;

                // Update gambar utama
                const mainImg = document.querySelector('.main-image img');
                mainImg.src = variant.img;
                mainImg.srcset = variant.img.replace('600x800', '400x533') + ' 400w, ' + variant.img + ' 600w, ' + variant.img.replace('600x800', '800x1067') + ' 800w';

                // Update tombol aktif
                document.querySelectorAll('.variant-btn').forEach(btn => {

                    btn.setAttribute('aria-pressed', 'false');
                });
                document.getElementById(`size-${size}`).classList.add('active');
                document.getElementById(`size-${size}`).setAttribute('aria-pressed', 'true');
                document.getElementById(`color-${color.split(' ')[0]}`).classList.add('active');
                document.getElementById(`color-${color.split(' ')[0]}`).setAttribute('aria-pressed', 'true');

                // Update URL
                const url = new URL(window.location);
                url.searchParams.set('size', size);
                url.searchParams.set('color', color);
